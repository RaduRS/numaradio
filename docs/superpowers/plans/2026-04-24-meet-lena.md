```tsx
import { AbsoluteFill, Img, staticFile, useCurrentFrame } from "remotion";
import { COLORS } from "../tokens/brand.ts";

/**
 * Pure — Ken Burns scale at a given frame. Linear interpolation from
 * zoomStart to zoomEnd over `durationFrames`, clamped at both ends.
 */
export function portraitScale(
  frame: number,
  durationFrames: number,
  zoomStart: number,
  zoomEnd: number,
): number {
  if (frame <= 0) return zoomStart;
  if (frame >= durationFrames) return zoomEnd;
  const t = frame / durationFrames;
  return zoomStart + (zoomEnd - zoomStart) * t;
}

export interface LenaPortraitProps {
  /** Duration of the Ken Burns zoom window, in frames. Default 120. */
  durationFrames?: number;
  /** Starting scale factor. Default 1.0. */
  zoomStart?: number;
  /** Ending scale factor. Default 1.05. */
  zoomEnd?: number;
  /** Teal color-grade overlay opacity [0,1]. Default 0. Set ~0.15 for subtle brand tint. */
  tealGrade?: number;
}

/**
 * Canonical Lena portrait with Ken Burns zoom and optional teal color grade.
 * Loads `staticFile("lena/lena-v1.png")`. Fills the composition via object-fit cover.
 */
export function LenaPortrait({
  durationFrames = 120,
  zoomStart = 1.0,
  zoomEnd = 1.05,
  tealGrade = 0,
}: LenaPortraitProps) {
  const frame = useCurrentFrame();
  const scale = portraitScale(frame, durationFrames, zoomStart, zoomEnd);

  return (
    <AbsoluteFill style={{ overflow: "hidden" }}>
      <Img
        src={staticFile("lena/lena-v1.png")}
        style={{
          width: "100%",
          height: "100%",
          objectFit: "cover",
          transform: `scale(${scale})`,
          transformOrigin: "center",
        }}
      />
      {tealGrade > 0 && (
        <AbsoluteFill
          style={{
            background: COLORS.accent,
            opacity: tealGrade,
            mixBlendMode: "overlay",
            pointerEvents: "none",
          }}
        />
      )}
    </AbsoluteFill>
  );
}
```

```tsx
import { AbsoluteFill, useCurrentFrame, interpolate } from "remotion";
import { FONTS } from "../tokens/brand.ts";

export interface ShowPanelProps {
  /** Show block name, e.g. "Night Shift". */
  name: string;
  /** Time range label, e.g. "00:00 — 05:00". */
  timeLabel: string;
  /** Short one-line description. */
  description: string;
  /** Full background color for the panel (hex). */
  backgroundColor: string;
  /** Accent color for typography highlights (hex). */
  accentColor: string;
}

/**
 * Full-screen parameterized panel used in MeetLena's four-show montage
 * (and in DayInNuma in Stage 2c.2). Hard-cut entrance; content fades in
 * over the first 10 frames of the sequence. No internal animation.
 */
export function ShowPanel({
  name,
  timeLabel,
  description,
  backgroundColor,
  accentColor,
}: ShowPanelProps) {
  const frame = useCurrentFrame();
  const contentOpacity = interpolate(frame, [0, 10], [0, 1], {
    extrapolateLeft: "clamp",
    extrapolateRight: "clamp",
  });

  return (
    <AbsoluteFill style={{ background: backgroundColor }}>
      <div
        style={{
          position: "absolute",
          inset: 0,
          display: "flex",
          flexDirection: "column",
          justifyContent: "center",
          alignItems: "center",
          gap: 36,
          padding: "0 80px",
          opacity: contentOpacity,
        }}
      >
        {/* Time range — mono, small, spaced */}
        <div
          style={{
            fontFamily: `"${FONTS.mono}", ui-monospace, monospace`,
            fontSize: 32,
            letterSpacing: "0.32em",
            color: accentColor,
            textTransform: "uppercase",
            opacity: 0.7,
          }}
        >
          {timeLabel}
        </div>

        {/* Show name — Archivo Black, huge */}
        <div
          style={{
            fontFamily: `"${FONTS.display}", system-ui, sans-serif`,
            fontSize: 180,
            lineHeight: 0.92,
            letterSpacing: "-0.02em",
            textTransform: "uppercase",
            color: accentColor,
            fontWeight: 800,
            fontStretch: "125%",
            textAlign: "center",
          }}
        >
          {name}
        </div>

        {/* Thin accent line */}
        <div
          style={{
            width: 160,
            height: 2,
            background: accentColor,
            opacity: 0.6,

            textAlign: "center",
            maxWidth: 900,
            opacity: 0.85,
          }}
        >
          {description}
        </div>
      </div>
    </AbsoluteFill>
  );
}
```

```tsx
// --- Music ducking ------------------------------------------------------

/**
 * Simpler ducking per spec's preferred alternative: duck flat across the
 * entire voice region (frame 110 to frame 1440), then ramp back up for
 * the payoff. Same perceptual result as per-segment pumping, cleaner code.
 */
function musicBedVolume(frame: number): number {
  // Fade-in 0-6
  if (frame < 6) return frame / 6;
  // Pre-voice sustain
  if (frame < 110) return 1.0;
  // Duck down (500ms)
  if (frame < 125) return 1.0 - ((frame - 110) / 15) * 0.75;
  // Ducked continuously while voice is active (across intro, montage, closer)
  if (frame < 1440) return 0.25;
  // Duck back up (500ms)
  if (frame < 1455) return 0.25 + ((frame - 1440) / 15) * 0.75;
  // Payoff sustain
  if (frame < 1740) return 1.0;
  // Fade-out (60 frames)
  if (frame < 1800) return 1.0 - (frame - 1740) / 60;
  return 0;
}

function PortraitBeat() {
  // Beats 1 + 2 combined: portrait fades in frames 0-30, holds, Ken Burns
  // zoom 1.0 → 1.10 over the full 660 frames. Waveform appears at bottom
  // during frames 120-660 (Lena monologue).
  const frame = useCurrentFrame();
  const portraitOpacity = interpolate(frame, [0, 30], [0, 1], {
    extrapolateLeft: "clamp",
    extrapolateRight: "clamp",

    extrapolateLeft: "clamp",
    extrapolateRight: "clamp",

  return (
    <AbsoluteFill>
      <div style={{ opacity: portraitOpacity }}>
        <LenaPortrait
          durationFrames={660}
          zoomStart={1.0}
          zoomEnd={1.10}
          tealGrade={0.12}
```
